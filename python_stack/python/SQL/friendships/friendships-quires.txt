

CREATE SCHEMA IF NOT EXISTS `friendships`;
USE `friendships`;

-- Create tables
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    first_name VARCHAR(255) NOT NULL,
    last_name VARCHAR(255) NOT NULL,
    created_at DATETIME DEFAULT NOW(),
    updated_at DATETIME DEFAULT NOW() ON UPDATE NOW()
);

CREATE TABLE friendships (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    friend_id INT NOT NULL,
    created_at DATETIME DEFAULT NOW(),
    updated_at DATETIME DEFAULT NOW() ON UPDATE NOW(),
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (friend_id) REFERENCES users(id)
);

-- MAIN QUERY: Display relationships
SELECT 
    u1.first_name AS first_name,
    u1.last_name AS last_name,
    u2.first_name AS friend_first_name,
    u2.last_name AS friend_last_name
FROM users u1
JOIN friendships f ON u1.id = f.user_id
JOIN users u2 ON f.friend_id = u2.id
ORDER BY u1.id;


-- NINJA QUERY 1: All friends of user 1
SELECT u2.first_name, u2.last_name
FROM friendships f
JOIN users u2 ON f.friend_id = u2.id
WHERE f.user_id = 1;


-- NINJA QUERY 2: Total count of friendships
SELECT COUNT(*) AS total_friendships FROM friendships;


-- NINJA QUERY 3: User with most friends
SELECT 
    u.first_name,
    u.last_name,
    COUNT(f.friend_id) AS friend_count
FROM users u
JOIN friendships f ON u.id = f.user_id
GROUP BY u.id, u.first_name, u.last_name
ORDER BY friend_count DESC
LIMIT 1;


-- NINJA QUERY 4: Friends of user 3 in alphabetical order
SELECT u2.first_name, u2.last_name
FROM friendships f
JOIN users u2 ON f.friend_id = u2.id
WHERE f.user_id = 3
ORDER BY u2.first_name ASC, u2.last_name ASC;
